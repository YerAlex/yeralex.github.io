1.3 背包、队列与栈

发表于 2019-03-18 | 更新于 2019-03-28 | 分类于 Algrs4 | 阅读次数:
| 字数统计: 3,692



本章主要介绍了基本的三种数据结构（背包、队列与栈），同时介绍了Java中的泛型与迭代特性，以及引入链表的概念。

背包是不考虑顺序往内部填充而且无法取出元素的数据结构，队列是先进先出，栈是先进后出。三者共有的操作是判断是否为空，输出内部元素个数以及向内部添加元素的操作。

基础数据结构分为顺序存储以及链式存储两种实现模式。下以栈为例。

首先考虑栈的定容实现：

import edu.princeton.cs.algs4.StdIn;
import edu.princeton.cs.algs4.StdOut;

public class FixedCapacityStackOfStrings {
    private String[] a;
    private int N;
    private int MAX_N;

    public FixedCapacityStackOfStrings(int cap) {
        a = new String[cap];
        MAX_N = cap;
    }

    public boolean isEmpty() {
        return N == 0;
    }

    public boolean isFull(){
        return N == MAX_N;
    }       // Solution for 1.3.1

    public int size() {
        return N;
    }

    public void push(String item) {
        a[N++] = item;
    }

    public String pop() {
        return a[--N];
    }

}

该方法由于不能扩充容量，所以可以对其改进而形成可变容栈：

import edu.princeton.cs.algs4.StdIn;

import java.util.Iterator;

public class DoublingStackOfStrings<Item> implements Iterable<Item>{
    private int N = 0 ;
    @SuppressWarnings("unchecked")
    private Item[] a = (Item[]) new Object[1];
    public boolean isEmpty(){
        return N == 0;
    }

    public int sizeArray(){
        return N;
    }
    public int sizeTotal(){
        return a.length;
    }
    private void resize(int max){
        @SuppressWarnings("unchecked")
        Item[] temp = (Item[]) new Object[max];
        for(int i = 0 ; i < N;i++){
            temp[i] = a[i];
        }
        a = temp;
    }
    public void push(Item item){
        if(N==a.length){
            resize(2*N);
        }
        a[N++] = item;
    }

    public Item pop(){
        Item item = a[--N];
        a[N] = null;
        if(N>0 && N==a.length/4)
            resize(a.length/2);
        return item;
    }

    public Iterator<Item> iterator(){
        return new ReverseArrayIterator();
    }
    private class ReverseArrayIterator implements Iterator<Item>{
        private int i = N;
        public boolean hasNext(){
            return i>0;
        }
        public Item next(){
            return a[--i];
        }
        public void remove(){}
    }

同时上部分引入了泛型与迭代的概念，泛型即在实现数据结构时用抽象类型代替相关数据类型，在生成时才指明其数据类型。迭代即通过迭代器使得可以迭代数据结构中的元素。

泛型的实现为在类名后添加<Item>，并在内部通过Item指代抽象数据类型。其中的例外是泛型数组的生成，需要使用形如(Item[]) new Object[len]的形式。

迭代首先需要添加调用函数import java.util.Iterator，其次在类声明中加入继承声明implements Iterable<Item>，Iterable类原有函数为iterator，该函数返回一个Iterator<Item>类。所以需要在函数内部声明

public Iterator<Item> iterator(){
    return new TempIterator();
}

同时上述TempIterator类需要继承Iterator类：

private TempIterator implements Iterator<Item>{

  1. 表头删除

介绍完基础链表操作后即可开始实现栈的链式存储模型

import java.util.Iterator;

public class Stack<Item> implements Iterable<Item> {
    private Node first;
    private int N;
    private class Node{
        Item item;
        Node next;
    }
    public boolean isEmpty() { return first == null;}
    public int size() { return N;}
    public void push(Item item){
        Node oldfirst = first;
        first = new Node();
        first.item = item;
        first.next = oldfirst;
        N++;
    }
    public Item pop(){
        Item item = first.item;
        first = first.next;
        N--;
        return item;
    }


    public Iterator<Item> iterator(){
        return new ListIterator();
    }
    private class ListIterator implements  Iterator<Item>{
        private  Node current = first;
        public boolean hasNext(){
            return current != null;
        }
        public void remove(){}
        public Item next(){
            Item item = current.item;
            current = current.next;
            return item;
        }
    }
}

以及队列的链式实现


import java.util.Iterator;

public class Queue<Item> implements Iterable<Item>{
        private Node first;
        private Node last;
        private int N;
        private class Node{
            Item item;
            Node next;
        }
        public boolean isEmpty(){ return  first == null; }
        public int size(){
            return N;
        }
        public void enqueue(Item item){
            Node oldlast = last;
            last = new Node();
            last.item = item;
            last.next = null;
            if(isEmpty()) first = last;
            else oldlast.next = last;
            N++;
        }

        public Item dequeue(){
            Item item = first.item;
            first = first.next;
            if(isEmpty()) last = null;
            N--;
            return item;
        }

        public Iterator<Item> iterator(){
            return new QueueArrayIterator();
        }

        private class QueueArrayIterator implements Iterator<Item>{
            private Node current = first;
            public boolean hasNext(){
                return current!=null;
            }
            public Item next(){
                Item item = current.item;
                current = current.next;
                return item;
            }
            public void remove(){}
        }
}

import java.util.Iterator;

public class CircleQueue<Item> implements Iterable<Item> {
    private class Node{
        Item item;
        Node next;
    }

    private Node last;
    CircleQueue(){
        last.next = last;
    }
    private int N = 0;
    public boolean isEmpty(){
        return last.next == last;
    }

    public int size(){
        return N;
    }

    public void enqueue(Item s){
        Node n_new = new Node();
        n_new.item = s;
        if(this.isEmpty()) {
            last.next = n_new;
        }
        else{
            Node current = last.next;
            while(current.next!=last)
                current = current.next;
            current.next=n_new;
        }
        n_new.next=last;
        N++;
    }

    public Item dequeue(){
        if(this.isEmpty()){
            return null;
        }
        else{
            Item it = last.next.item;
            last.next=last.next.next;
            N--;
            return it;
        }
    }

    public Iterator<Item> iterator(){
        return new CircleQueueIterator();
    }

    private class CircleQueueIterator implements Iterator<Item>{
        private Node current = last.next;
        public boolean hasNext(){
            return current.next!=last;
        }
        public void remove(){}
        public Item next(){
            Item it = current.item;
            current=current.next;
            return it;
        }
    }

}

import java.util.Iterator;

public class ResizingArrayDeque<Item> implements Iterable<Item> {
    private Item[] a;
    private int N;
    private int first;

    @SuppressWarnings("unchecked")
    public ResizingArrayDeque(){
        a = (Item[]) new Object[1];
        N = 0;
        first = 0;
    }
    public boolean isEmpty(){
        return N == 0;
    }

    public int size(){
        return N;
    }

    public int count(){
        return a.length;
    }

    @SuppressWarnings("unchecked")
    private void resizeRight(int num){
        Item[] temp = (Item[]) new Object[num];
        for(int i = 0;i<N;i++){
            temp[first+i] = a[first+i];
        }
        a = temp;
    }

    @SuppressWarnings("unchecked")
    private void resizeLeft(int num){
        Item[] temp = (Item[]) new Object[num];
        for(int i = 0;i<N;i++){
            temp[num+first-a.length+i] = a[first+i];
        }
        first = num-a.length+first;
        a =temp;

    }

    public void pushLeft(Item item){
        if(first==0){
            resizeLeft(a.length+N+1);
            a[--first] = item;
            N++;
        }
        else{
            a[--first] = item;
            N++;
        }
    }

    public void pushRight(Item item){
        if(first+N == a.length){
            resizeRight(a.length+N);
            N++;
            a[first+N-1] = item;
        }
        else{
            N++;
            a[first+N-1] = item;
        }
    }

    public Item popLeft(){
        if(N==0)
            return null;
        else{
            Item it = a[first++];
            N--;
            if(N>0 && N==(first+N)/4){
                resizeLeft(a.length-(first+N)/2);
            }
            return it;
        }
    }

    public Item popRight(){
        if(N==0)
            return null;
        else{
            Item it = a[first+N-1];
            N--;
            if(N>0 && N ==(a.length-first)/4){
                resizeRight(a.length-(a.length-first)/2);
            }
            return it;
        }
    }


    public Iterator<Item> iterator(){
        return new DequeIterator();
    }

    private class DequeIterator implements Iterator<Item>{
        private int current = first;
        public boolean hasNext(){
            return current<(first+N);
        }
        public void remove(){}
        public Item next(){
            Item it = a[current];
            current++;
            return it;
        }
    }
    public static void main(String[] args) {
        ResizingArrayDeque<String> deque = new ResizingArrayDeque<String>();
        deque.pushLeft("c");
        System.out.println("deque size: " + deque.size() +"| deque length: " + deque.count());
        deque.pushLeft("b");
        System.out.println("deque size: " + deque.size() +"| deque length: " + deque.count());
        deque.pushRight("d");

        deque.pushRight("e");
        System.out.println("deque size: " + deque.size() +"| deque length: " + deque.count());
        Iterator<String> it = deque.iterator();
        while (it.hasNext()) {
            System.out.println(it.next());
        }

        while (!deque.isEmpty()) {

1.3.45

这道题可能和题目给出的要求有一些出入，我的实现结果是给定一个序列，判断能否通过对一个栈进行出栈、入栈操作来实现该序列。在进行判断的同时，输出实现该序列的步骤。

（个人感觉自己这道题实现的很不好，有很多重if判断的嵌套，可读性很差，而且实现起来也不简洁，逻辑关系混乱。但是经过测试还是能用的。。。）

/* *****************************************************************************
 *  Name: StackGeneration
 *  Date: 2019/03/28
 *  Description: Show whether a given sequence can be realized by using push and pop operation of a stack
 *  Index of Problems: 1.3.45
 **************************************************************************** */

import edu.princeton.cs.algs4.StdOut;

public class StackGeneration {

    public boolean downOverflow(String[] seqs){
        int cnt = 0;
        for(int i = 0;i<seqs.length;i++){
            if(seqs[i].equals("-")){
                cnt--;
                if(cnt<0)
                    return false;
            }
            else
                cnt++;
        }
        return true;
    }

    public boolean GeneratePossibility(String[] seq){
        boolean flg = true;
        Queue<String> rslt = new Queue<String>();
        Queue<Integer> q = new Queue<Integer>();
        Stack<Integer> tempStack = new Stack<Integer>();
        int tempQ=0;
        int td = 0;
        for(int i =0; i< seq.length;i++)
            q.enqueue(i);
        for(int i=0;i<seq.length;i++){
            int d_seq = Integer.parseInt(seq[i]);
            if(q.isEmpty()){
                if(tempStack.pop()!= d_seq){
                    flg = false;
                    return flg;
                }
                else
                    StdOut.print("- ");
            }
            else{
                if(tempStack.isEmpty()){
                    tempQ = q.dequeue();
                    tempStack.push(tempQ);
                    StdOut.print(tempQ + " ");
                }
                    td = tempStack.pop();
                    if(td == d_seq){
                        StdOut.print("- ");
                    }
                    else{
                        if(td>d_seq){
                            flg = false;
                            return flg;
                        }
                        else{
                            tempQ = q.dequeue();
                            if(tempQ > d_seq){
                                flg = false;
                                return flg;
                            }
                            else{
                                tempStack.push(td);
                                if(tempQ == d_seq) {
                                    StdOut.print(tempQ + " - ");
                                }
                                else {
                                    while (tempQ < d_seq) {
                                        tempStack.push(tempQ);
                                        StdOut.print(tempQ + " ");
                                        tempQ = q.dequeue();
                                    }
                                    StdOut.print(tempQ + " ");
                                    StdOut.print("- ");
                                }
                            }
                        }
                    }
                }
            }

        return flg;
    }
    public static void main(String[] args) {
        String[] a = {"2","1","4","3","6","5","8","7","9","0"};
        StackGeneration test = new StackGeneration();
        StdOut.print(test.GeneratePossibility(a));
    }
}
